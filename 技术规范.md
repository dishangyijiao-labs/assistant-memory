# IDE-Memory 项目技术规范

## 项目概述

IDE-Memory 是一个用于统一管理和记录 IDE 中与 AI 聊天工具会话内容的应用程序。它支持记录和管理 GitHub Copilot、Cursor、VS Code、Cloud Code、CodeX 和 Gemini 等 AI 聊天工具的聊天历史和会话信息。

### 1. 后端规范

#### 1.1 Python 代码规范

```python
# 使用 Flask 和 SQLAlchemy 的示例代码结构

from flask import Flask
from flask_sqlalchemy import SQLAlchemy
from datetime import datetime

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///ide_memory.db'
app.config['SQLALCHEMY_TRACK_MODIFICATIONS'] = False

db = SQLAlchemy(app)

class Session(db.Model):
    """会话模型"""
    id = db.Column(db.Integer, primary_key=True)
    tool_name = db.Column(db.String(100), nullable=False)
    tool_version = db.Column(db.String(50))
    workspace_path = db.Column(db.String(500))
    project_name = db.Column(db.String(200))
    session_type = db.Column(db.String(50))
    created_at = db.Column(db.DateTime, default=datetime.utcnow)
    updated_at = db.Column(db.DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)
    
    messages = db.relationship('Message', backref='session', lazy=True)
    
    def to_dict(self):
        """将模型转换为字典"""
        return {
            'id': self.id,
            'tool_name': self.tool_name,
            'tool_version': self.tool_version,
            'workspace_path': self.workspace_path,
            'project_name': self.project_name,
            'session_type': self.session_type,
            'created_at': self.created_at.isoformat(),
            'updated_at': self.updated_at.isoformat()
        }

class Message(db.Model):
    """消息模型"""
    id = db.Column(db.Integer, primary_key=True)
    session_id = db.Column(db.Integer, db.ForeignKey('session.id'), nullable=False)
    role = db.Column(db.String(20), nullable=False)
    content = db.Column(db.Text, nullable=False)
    timestamp = db.Column(db.DateTime, default=datetime.utcnow)
    
    code_snippets = db.relationship('CodeSnippet', backref='message', lazy=True)
    
    def to_dict(self):
        """将模型转换为字典"""
        return {
            'id': self.id,
            'session_id': self.session_id,
            'role': self.role,
            'content': self.content,
            'timestamp': self.timestamp.isoformat()
        }

if __name__ == '__main__':
    db.create_all()
    app.run(debug=True)
```

#### 1.2 API 设计规范

```python
# API 路由设计示例
from flask import Flask, request, jsonify
from flask_sqlalchemy import SQLAlchemy
from models import Session, Message

app = Flask(__name__)

@app.route('/api/sessions', methods=['GET'])
def get_sessions():
    """获取所有会话"""
    try:
        sessions = Session.query.all()
        return jsonify([session.to_dict() for session in sessions])
    except Exception as e:
        return jsonify({'error': str(e)}), 500

@app.route('/api/sessions', methods=['POST'])
def create_session():
    """创建新会话"""
    try:
        data = request.get_json()
        new_session = Session(**data)
        db.session.add(new_session)
        db.session.commit()
        return jsonify(new_session.to_dict()), 201
    except Exception as e:
        db.session.rollback()
        return jsonify({'error': str(e)}), 400

@app.route('/api/sessions/<int:id>', methods=['GET'])
def get_session(id):
    """获取单个会话"""
    try:
        session = Session.query.get_or_404(id)
        return jsonify(session.to_dict())
    except Exception as e:
        return jsonify({'error': str(e)}), 404

if __name__ == '__main__':
    app.run(debug=True)
```

### 2. 前端规范

#### 2.1 React 组件规范

```typescript
// SessionList 组件示例
import React, { useEffect, useState } from 'react';
import { getSessions, deleteSession, Session } from '../services/sessionApi';
import SessionItem from './SessionItem';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Search } from 'lucide-react';

const SessionList: React.FC = () => {
  const [sessions, setSessions] = useState<Session[]>([]);
  const [searchQuery, setSearchQuery] = useState('');
  const [loading, setLoading] = useState(false);

  const loadSessions = async () => {
    try {
      setLoading(true);
      const data = await getSessions();
      setSessions(data);
    } catch (error) {
      console.error('Failed to load sessions:', error);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    loadSessions();
  }, []);

  const filteredSessions = sessions.filter(session =>
    session.project_name?.toLowerCase().includes(searchQuery.toLowerCase()) ||
    session.tool_name?.toLowerCase().includes(searchQuery.toLowerCase())
  );

  return (
    <div className="space-y-4">
      <div className="flex justify-between items-center">
        <h2 className="text-2xl font-bold">会话列表</h2>
        <div className="flex items-center space-x-2">
          <Input
            type="text"
            placeholder="搜索会话..."
            value={searchQuery}
            onChange={(e) => setSearchQuery(e.target.value)}
            className="max-w-md"
          />
          <Button onClick={loadSessions} disabled={loading}>
            {loading ? '加载中...' : '刷新'}
          </Button>
        </div>
      </div>

      {loading ? (
        <div className="text-center py-8">加载中...</div>
      ) : filteredSessions.length === 0 ? (
        <div className="text-center py-8 text-gray-500">
          {searchQuery ? '未找到匹配的会话' : '暂无会话'}
        </div>
      ) : (
        <div className="grid gap-4">
          {filteredSessions.map(session => (
            <SessionItem
              key={session.id}
              session={session}
              onDelete={loadSessions}
            />
          ))}
        </div>
      )}
    </div>
  );
};

export default SessionList;
```

#### 2.2 API 服务规范

```typescript
// API 服务示例
import axios from 'axios';
import { Session, Message, Tag } from '../types';

const API_BASE_URL = import.meta.env.VITE_API_BASE_URL || 'http://localhost:5000/api';

const api = axios.create({
  baseURL: API_BASE_URL,
  headers: {
    'Content-Type': 'application/json',
  },
});

// 会话 API
export const getSessions = async (): Promise<Session[]> => {
  const response = await api.get('/sessions');
  return response.data;
};

export const getSession = async (id: number): Promise<Session> => {
  const response = await api.get(`/sessions/${id}`);
  return response.data;
};

export const createSession = async (session: Omit<Session, 'id' | 'created_at' | 'updated_at'>): Promise<Session> => {
  const response = await api.post('/sessions', session);
  return response.data;
};

export const updateSession = async (id: number, session: Partial<Session>): Promise<Session> => {
  const response = await api.put(`/sessions/${id}`, session);
  return response.data;
};

export const deleteSession = async (id: number): Promise<void> => {
  await api.delete(`/sessions/${id}`);
};

// 消息 API
export const getMessages = async (sessionId: number): Promise<Message[]> => {
  const response = await api.get(`/sessions/${sessionId}/messages`);
  return response.data;
};

export const createMessage = async (message: Omit<Message, 'id' | 'timestamp'>): Promise<Message> => {
  const response = await api.post(`/sessions/${message.session_id}/messages`, message);
  return response.data;
};

// 标签 API
export const getTags = async (sessionId?: number): Promise<Tag[]> => {
  const url = sessionId ? `/sessions/${sessionId}/tags` : '/tags';
  const response = await api.get(url);
  return response.data;
};

export const addTag = async (sessionId: number, tag: Omit<Tag, 'id'>): Promise<Tag> => {
  const response = await api.post(`/sessions/${sessionId}/tags`, tag);
  return response.data;
};

export default api;
```

### 3. 数据库规范

#### 3.1 数据库初始化脚本

```python
# 数据库初始化脚本示例
from backend.src.utils.database import db
from backend.src.models.session import Session
from backend.src.models.message import Message
from backend.src.models.tag import Tag
from backend.src.models.snippet import CodeSnippet
from datetime import datetime

def init_database():
    """初始化数据库"""
    try:
        db.create_all()
        
        # 初始化一些示例数据（可选）
        session1 = Session(
            tool_name='GitHub Copilot',
            tool_version='v1.23.45',
            workspace_path='/home/user/projects/ide-memory',
            project_name='IDE-Memory',
            session_type='chat'
        )
        
        session2 = Session(
            tool_name='Cursor',
            tool_version='v0.2.1',
            workspace_path='/home/user/projects/ide-memory',
            project_name='IDE-Memory',
            session_type='chat'
        )
        
        message1 = Message(
            session_id=1,
            role='user',
            content='如何使用 Python 读取 Excel 文件？'
        )
        
        message2 = Message(
            session_id=1,
            role='assistant',
            content='您可以使用 pandas 库来读取 Excel 文件，示例代码如下：'
        )
        
        snippet = CodeSnippet(
            message_id=2,
            code='import pandas as pd\ndf = pd.read_excel("file.xlsx")\nprint(df.head())',
            language='python',
            filename='example.py'
        )
        
        tag1 = Tag(session_id=1, tag='Python')
        tag2 = Tag(session_id=1, tag='Excel')
        tag3 = Tag(session_id=2, tag='TypeScript')
        
        db.session.add_all([session1, session2, message1, message2, snippet, tag1, tag2, tag3])
        db.session.commit()
        
        print('数据库初始化成功！')
    except Exception as e:
        db.session.rollback()
        print(f'数据库初始化失败：{str(e)}')

if __name__ == '__main__':
    from backend.src.main import app
    with app.app_context():
        init_database()
```
